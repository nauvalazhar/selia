# Command

An autocomplete inside a dialog for quick interaction.

<Preview name="basic" />

## Anatomy

Import the component and assemble its parts:

```tsx filename="Anatomy"
<Command>
  <CommandTrigger />
  <CommandContent>
    <CommandBody>
      <Autocomplete>
        <AutocompleteInput />
        <AutocompleteTrigger />
        <AutocompleteIcon />
        <AutocompleteClear />
        <AutocompleteValue />
        <AutocompletePopup>
          <AutocompleteEmpty />
          <AutocompleteList>
            <AutocompleteRow>
              <AutocompleteItem />
            </AutocompleteRow>
            <AutocompleteSeparator />
            <AutocompleteGroup>
              <AutocompleteGroupLabel />
            </AutocompleteGroup>
            <AutocompleteCollection />
          </AutocompleteList>
        </AutocompletePopup>
      </Autocomplete>
    </CommandBody>
    <CommandFooter>
      <CommandFooterItem />
      <CommandFooterText />
    </CommandFooter>
  </CommandContent>
</Command>
```
