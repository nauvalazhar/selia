## Dependencies

First, install the dependencies:

<CodeTabs items={['npm', 'yarn', 'pnpm', 'bun']}>
  <CodeTabsPanel value="npm" language="bash">
    npm install @base-ui/react tailwind-merge clsx class-variance-authority
  </CodeTabsPanel>
  <CodeTabsPanel value="yarn" language="bash">
    yarn add @base-ui/react tailwind-merge clsx class-variance-authority
  </CodeTabsPanel>
  <CodeTabsPanel value="pnpm" language="bash">
    pnpm add @base-ui/react tailwind-merge clsx class-variance-authority
  </CodeTabsPanel>
  <CodeTabsPanel value="bun" language="bash">
    bun add @base-ui/react tailwind-merge clsx class-variance-authority
  </CodeTabsPanel>
</CodeTabs>
