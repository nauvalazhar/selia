# Toast

Brief notification messages that appear temporarily at screen top.

<Preview name="basic" />

## Anatomy

Import the component and assemble its parts:

```tsx filename="Anatomy"
<Toast />
```

## Usage

Add the `Toast` component to your top-level component such as `root.tsx` or `layout.tsx`.

```tsx filename="root.tsx"
export function Layout({ children }: { children: React.ReactNode }) {
  return (
    <html lang="en">
      <head>
        <meta charSet="utf-8" />
        <meta name="viewport" content="width=device-width, initial-scale=1" />
        <Meta />
        <Links />
      </head>
      <body className="bg-background">
        <div className="root">{children}</div>
// [!code highlight:1]
        <Toast />
        <ScrollRestoration />
        <Scripts />
      </body>
    </html>
  );
}
```

To create a toast, use the `toastManager` object.

```tsx filename="root.tsx"
import { toastManager } from 'components/selia/toast';

toastManager.create({
  title: 'Toast',
  description: 'This is a toast',
  type: 'success',
});
```

Or use the `anchoredToastManager` object to create an anchored toast.

```tsx filename="root.tsx"
import { anchoredToastManager } from 'components/selia/toast';

anchoredToastManager.create({
  title: 'Toast',
  description: 'This is a toast',
  type: 'success',
  positionerProps: {
    anchor: buttonRef.current,
    sideOffset: 8,
  },
});
```
