## Usage

### Import

Import the component and its related parts.

```ts filename="select Parts"
import {
  Select,
  SelectTrigger,
  SelectValue,
  SelectPopup,
  SelectList,
  SelectItem,
  SelectGroup,
  SelectGroupLabel,
  SelectSeparator
} from "components/selia/select";
```

### Structure

The expected structure and composition of the component.

```tsx filename="Anatomy"
<Select>
  <SelectTrigger>
    <SelectValue />
  </SelectTrigger>
  <SelectPopup>
    <SelectList>
      <SelectItem>Item 1</SelectItem>
      <SelectItem>Item 2</SelectItem>
      <SelectItem>Item 3</SelectItem>
    </SelectList>
  </SelectPopup>
</Select>
```
